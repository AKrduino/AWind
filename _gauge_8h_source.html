89  float factor=1000/(_maxValue-_minValue);
90  float dc_val=1000/(_maxValue-_minValue)*(_value-_minValue);
91  float step_val=(_maxValue-_minValue)/(_numTicks-1);
92  switch(_gaugeType)
93  {
94  case Bar:
95  dc->FillRoundRect(0,0,Width(),Height());
96  break;
97  case RadialPointer:
98  {
99  int radius=Height()*1;
100  int x0=Width()/2;
101  int y0=Height()*1.4;
102  float sec_angle_rad=3.14/2*0.8;
103  dc->DrawRoundRect(0,0,Width(),Height());
104  dc->Sector(x0,y0,radius,sec_angle_rad);
105  float angle_step=sec_angle_rad/(_numTicks-1.0);
106  float angle;
107  float sin_val;
108  float cos_val;
109  float value;
110  int x_offset;
111  for(int i=0;i<_numTicks;i++)
112  {
113  angle=-sec_angle_rad/2.0+angle_step*i;
114  sin_val=sin(angle);
115  cos_val=cos(angle);
116  dc->MoveTo(x0+radius*sin_val,y0-radius*cos_val);
117  dc->LineTo(x0+radius*1.1*sin_val,y0-radius*1.1*cos_val);
118  value=_minValue+step_val*i;
119  x_offset=(value==0?0:log10(value))+3;
120  if(value<0)
121  x_offset++;
122  if(i==0)
123  x_offset=0;
124  else if(i == _numTicks-1)
125  x_offset=-(x_offset-0.5)*dc->FontWidth();
126  else
127  x_offset=-x_offset/2*dc->FontWidth();
128  dc->DrawNumber(value,1,x0+radius*1.1*sin_val+x_offset,y0-radius*1.1*cos_val-(dc->FontHeight()*(i==0||i==_numTicks-1?1.5:1)));
129  }
130  angle=-sec_angle_rad/2.0+sec_angle_rad/1000.0*dc_val;
131  sin_val=sin(angle);
132  cos_val=cos(angle);
133  dc->MoveTo(x0+radius*0.95*sin_val,y0-radius*0.95*cos_val);
134  dc->LineTo(x0+radius*0.7*sin_val,y0-radius*0.7*cos_val);
135  break;
136  }
