107  void OnDraw(DC *dc)
108  {
109  dc->SetColor(_color);
110  dc->SetFont(SmallFont);
111  float range=_maxValue-_minValue;
112  float step_val=(range)/(_numTicks-1);
113  switch(_gaugeType)
114  {
115  case Bar:
116  {
117  const int tick_length=5;
118  int right_offset=Width()-(AHelper::GetNumberLength(max(_minValue,_maxValue),1)*dc->FontWidth()+dc->FontWidth()+tick_length);
119  if(!_drawOnlyPointer)
120  {
121  dc->DrawRoundRect(0,0,right_offset,Height());
122  int y;
123  int scale_step=Height()/(_numTicks-1);
124  for(int i=0;i<_numTicks;i++)
125  {
126  y=Height()-scale_step*i;
127  dc->MoveTo(right_offset,y);
128  dc->LineTo(right_offset+tick_length,y);
129  dc->DrawNumber(_minValue+i*step_val,1,right_offset+tick_length+dc->FontWidth(),y-(dc->FontHeight()*(i==_numTicks-1?0:1)));
130  }
131  }
132  if(_oldValue>_value)
133  {
134  dc->SetColor(GetBackColor());
135  dc->FillRoundRect(2,1,right_offset-2,min(Height()-2,Height()-_value*Height()/range+2));
136  }
137  dc->SetColor(Color::Red);
138  dc->FillRoundRect(2,Height()-_value*Height()/range,right_offset-2,Height()-1);
139  break;
140  }
141  case RadialPointer:
142  {
143  float factor=1000/range;
144  int radius=Height()*1;
145  int x0=Width()/2;
146  int y0=Height()*1.4;
147  if(!_drawOnlyPointer)
148  {
149  dc->DrawRoundRect(0,0,Width(),Height());
150  dc->Sector(x0,y0,radius,_sector_angle_rad);
151  float angle_step=_sector_angle_rad/(_numTicks-1.0);
152  float angle;
153  float sin_val;
154  float cos_val;
155  float value;
156  int x_offset;
157  for(int i=0;i<_numTicks;i++)
158  {
159  angle=-_sector_angle_rad/2.0+angle_step*i;
160  sin_val=sin(angle);
161  cos_val=cos(angle);
162  dc->MoveTo(x0+radius*sin_val,y0-radius*cos_val);
163  dc->LineTo(x0+radius*1.1*sin_val,y0-radius*1.1*cos_val);
164  value=_minValue+step_val*i;
165  x_offset=AHelper::GetNumberLength(value,1);
166  if(value<0)
167  x_offset++;
168  if(i==0)
169  x_offset=0;
170  else if(i == _numTicks-1)
171  x_offset=-(x_offset-0.5)*dc->FontWidth();
172  else
173  x_offset=-x_offset/2*dc->FontWidth();
174  dc->DrawNumber(value,1,x0+radius*1.1*sin_val+x_offset,y0-radius*1.1*cos_val-(dc->FontHeight()*(i==0||i==_numTicks-1?1.5:1)));
175  }
176  }
177  DrawPointer(_oldValue,x0,y0,radius,dc,GetBackColor());
178  DrawPointer(_value,x0,y0,radius,dc,_color);
179  break;
180  }
181  }
182  }
183 private:
185  void DrawPointer(float value,int x0,int y0,int radius,DC *dc,Color color)
186  {
187  dc->SetColor(color);
188  float dc_val=1000/(_maxValue-_minValue)*(value-_minValue);
189  float angle=-_sector_angle_rad/2.0+_sector_angle_rad/1000.0*dc_val;
190  float sin_val=sin(angle);
191  float cos_val=cos(angle);
192  dc->MoveTo(x0+radius*0.95*sin_val,y0-radius*0.95*cos_val);
193  dc->LineTo(x0+radius*0.7*sin_val,y0-radius*0.7*cos_val);
194 
195  }
