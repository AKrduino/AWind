74  _textValue = new TextBoxNumber(offset,offset,Width(),1,_sensorManager->Sensor()->Precission(),Color::White);

104  virtual bool OnTouch(int x, int y)
105  {
106  if(_mode == ChartHowr)
107  {
108  _textValue->SetVisible(true);
109  _textName->SetVisible(true);
110  _chartWnd->SetVisible(false);
111  _textChartAxis->SetVisible(false);
112  _mode=Text;
113  }
114  else if(_mode == Text)
115  {
116  _textValue->SetVisible(false);
117  _textName->SetVisible(false);
118  _chartWnd->SetVisible(true);
119  _textChartAxis->SetVisible(true);
120  _chartWnd->SetBuffer(_sensorManager->SecBuffer());
121  _textChartAxis->SetText(F("sec"));
122  _mode=ChartSec;
123  }
124  else if(_mode == ChartSec)
125  {
126  _chartWnd->SetBuffer(_sensorManager->MinBuffer());
127  _textChartAxis->SetText(F("min"));
128  _mode=ChartMin;
129  }
130  else if(_mode == ChartMin)
131  {
132  _chartWnd->SetBuffer(_sensorManager->HowrsBuffer());
133  _textChartAxis->SetText(F("howr"));
134  _mode=ChartHowr;
135  }
136  _textChartAxis->SetVisible(!_textValue->IsVisible());
137  Invalidate();
138  return true;
139  }
140  void NotifySensorHasData(SensorManager *sensorManager)
141  {
142  _textValue->SetStatus(_sensorManager->Status()!=Error);
143  float value=_sensorManager->GetData();
144  if(_sensorManager->Status()!=Error)
145  {
146  if(_sensorManager->Status() == ApplicationAlarm)
147  Window::SetBackColor(Color::Red);
148  else
149  Window::SetBackColor(_normalBkColor);
150  }
151  if(_sensorManager->Status()!=Error && _textValue->GetNumber()!=value)
152  {
153  _textValue->SetNumber(value);
154  _mode == Text?Invalidate():_chartWnd->Invalidate();
155  }
156  else if(_sensorManager->Status()!=Error && _mode == ChartSec)
157  _chartWnd->Invalidate();
158  }
