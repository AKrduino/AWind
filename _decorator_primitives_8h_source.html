177  int EstimateRight(DC *dc)
178  {
179  dc->SetFont(_font);
180  switch(_orientation)
181  {
182  case VerticalRight:
183  case VerticalLeft:
184  return (AHelper::GetNumberLength(max(_minValue,_maxValue),1)*dc->FontWidth()+_tick_length)+_offsetX;
185  case HorizontalTop:
186  case HorizontalBottom:
187  return _length+_offsetX;
188  }

191  {
192  return _offsetY;
193  }
194  int EstimateBottom(DC *dc)
195  {
196  dc->SetFont(_font);
197  switch(_orientation)
198  {
199  case VerticalRight:
200  case VerticalLeft:
201  return _length+_offsetY;
202  case HorizontalTop:
203  case HorizontalBottom:
204  return _offsetY+dc->FontHeight()+_tick_length;
205  }
206  }
207  void Draw(DC *dc,int left,int top,int width,int height)
208  {
209  dc->SetFont(_font);
210  float range=_maxValue-_minValue;
211  float step_val=(range)/(_numTicks-1);
212  switch(_orientation)
213  {
214  case VerticalLeft:
215  case VerticalRight:
216  {
217  int y;
218  float scale_step=_length/(_numTicks-1.0);
219  int axis_right=EstimateRight(dc);
220  for(int i=0;i<_numTicks;i++)
221  {
222  y=_length+_offsetY-scale_step*i;
223  float value=_minValue+i*step_val;
224  if(_orientation == VerticalRight)
225  {
226  dc->MoveTo(_offsetX,y);
227  dc->LineTo(_offsetX+_tick_length,y);
228  dc->DrawNumber(value,1,_offsetX+_tick_length,y-(dc->FontHeight()*(i==_numTicks-1?0:1)));
229  }
230  else
231  {
232  dc->MoveTo(axis_right,y);
233  dc->LineTo(axis_right-_tick_length,y);
234  int val_width=AHelper::GetNumberLength(value,1)*dc->FontWidth();
235  dc->DrawNumber(value,1,axis_right-val_width,y-(dc->FontHeight()*(i==_numTicks-1?0:1)));
236  }
237  }
238  }
239  }
240  }
