134  out<<F("touch in gauges top")<<endln;
135  if(_sensorManager!=NULL)

137  }
138  else if(window==_btnBottom)
139  {
140  out<<F("touch in gauges bottom")<<endln;
141  if(_sensorManager!=NULL)

143  }
144  }
